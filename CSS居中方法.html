    CSS居中方法

    临风 3/21/2022 CSS居中

    # 行内元素

    <div class="parent">
    <span class="child">content</span>
</div>

    # 水平居中

    # text-align: center;

    .parent {
	background-color: red;
	text-align: center;
}

    # fit-content

    .parent {
	background-color: red;
    width: fit-content; //此时父元素的宽度就会适应子元素的宽度
    margin: auto； //实现水平居中的效果
}

    # 垂直居中

    # line-height：只适用于单行文本

    .parent {
	background-color: red;
	height: 200px; 
	line-height: 200px//数值需与height高度一样
}

    # 块级元素

    <div class="parent">
    <div class="child"></div>
</div>

    # 水平居中

    # margin: 0 auto;

    .parent {
	background-color: red;
}
.child {
	width: 100px;
	margin: 0 auto; //使内边距左右自适应
	background-color: blue;
}

    # 水平垂直居中

    # 1. 定位 (需知道子元素的宽高)

    .parent {
	position: relative;
	height: 200px;
	background-color: red;
}
.child {
	position: absolute;
	width: 100px;
	height: 100px;
	background: blue;
	left: 50%;
	top: 50%;
	margin-top: -50px;
	margin-left: -50px;
	
}

    .parent {
	position: relative;
	height: 200px;
	background-color: red;
}
.child {
	position: absolute;
	width: 100px;
	height: 100px;
	background: blue;
	left: calc(50% - 50px);
	top: calc(50% - 50px);
}

    # 2. 定位 + transform(不需知道子元素的宽高)

    .parent {
	position: relative;
	height: 200px;
	background-color: red;
}
.child {
	position: absolute;
	width: 100px;
	height: 100px;
	background: blue;
	left: 50%;
	top: 50%;
	transform: translate(-50%, -50%);
}

    # 3. 定位 + margin

    .parent {
	position: relative;
	height: 200px;
	background-color: red;
}
.child {
	position: absolute;
	width: 100px;
	height: 100px;
	background: blue;
	left: 0;
	top: 0;
	right: 0;
	bottom: 0; //此时，子元素会填充父元素的所有可用空间，所以在水平垂直方向上就有了可分配的空间
	margin: auto;
}

    # 4. Padding

    .parent {   //父元素没有设置宽高，此时其宽高就是子元素的宽高
	background-color: red; 
	padding: 20px; 
}
.child {
	height: 200px;
}

    # 5.flex布局

    .parent {
	height: 200px;
	background-color: red; 
	display: flex;
	align-items: center;
	jusfiy-content: center;
}
.child {
	width: 100px;
	height: 100px;
	background-color: blue;
}

    # 伪元素

    .parent {
	height: 200px;
	text-align: center; //只对行内元素生效
	background-color: red; 
}
.child {
	width: 100px;
	height: 100px;
	display: inline-block; 
	vertical-align: middle;
	background-color: blue;
}
.parent::before { //before伪元素默认为行内元素
	content: "";
	width: 20px;
	height: 200px;
	display: inline-block; 
	vertical-align: middle;
}

    # calc宽高相等

    .parent {
	height: 600px;
	width: 600px
	background-color: red; 
}
.child {
	width: 100px;
	height: 100px;
	padding: calc((100% - 100px) / 2); //给子元素添加内边距，使得子元素大小父元素相等
	background-clip: content-box;//让背景颜色只对content有效，而不对padding生效
	background-color: blue;
}
